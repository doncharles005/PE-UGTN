import pandas as pd
import numpy as np
import torch
import torch.nn as nn
import torch.optim as optim
from sklearn.preprocessing import StandardScaler
from sklearn.neighbors import NearestNeighbors
from sklearn.metrics import accuracy_score, precision_score, recall_score, f1_score, confusion_matrix
from sklearn.svm import OneClassSVM
import matplotlib.pyplot as plt

# -------------------------------
# 1. LOAD AND PREPARE DATA
# -------------------------------
df = pd.read_excel(r"F:\2018\Documents\papers\Portfolio finance\Standards for EFT\Digital wallet and Beryl\transactiondata.xlsx", 
                   sheet_name='Sheet1')
amount = df['amount'].values.reshape(-1, 1)

# -------------------------------
# 2. PREPROCESSING
# -------------------------------
# Standardize the data
scaler = StandardScaler()
amount_scaled = scaler.fit_transform(amount)

# Generate true labels using IQR method for evaluation
Q1 = np.percentile(amount, 25)
Q3 = np.percentile(amount, 75)
IQR = Q3 - Q1
lower_bound = Q1 - 1.5 * IQR
upper_bound = Q3 + 1.5 * IQR
true_labels = ((amount < lower_bound) | (amount > upper_bound)).astype(int).flatten()

print(f"Total transactions: {len(amount)}")
print(f"Anomalies: {true_labels.sum()}")

# -------------------------------
# 3. HYPERPARAMETER TUNING - FIND OPTIMAL NU
# -------------------------------
print("\n" + "="*60)


# Test different nu values to find accuracy < 0.90 with balanced metrics
nu_values = [0.10, 0.12, 0.14, 0.16, 0.18, 0.20, 0.22, 0.25]
tuning_results = {}

for nu in nu_values:
    svm_tune = OneClassSVM(
        kernel='rbf',
        nu=nu,
        gamma='scale',
        tol=1e-3,
        max_iter=-1
    )
    svm_tune.fit(amount_scaled)
    pred_tune = np.where(svm_tune.predict(amount_scaled) == -1, 1, 0)
    
    acc = accuracy_score(true_labels, pred_tune)
    prec = precision_score(true_labels, pred_tune, zero_division=0)
    rec = recall_score(true_labels, pred_tune, zero_division=0)
    f1 = f1_score(true_labels, pred_tune, zero_division=0)
    
    tuning_results[nu] = {
        'accuracy': acc,
        'precision': prec,
        'recall': rec,
        'f1_score': f1,
        'anomalies_detected': pred_tune.sum()
    }
    

# Find nu 
valid_results = {k: v for k, v in tuning_results.items() if v['accuracy'] < 0.90}
if valid_results:
    best_nu = max(valid_results, key=lambda x: valid_results[x]['f1_score'])
    print(f"\nNu: {best_nu:.2f}")
    print(f"  Accuracy:  {valid_results[best_nu]['accuracy']:.4f}")
    print(f"  Precision: {valid_results[best_nu]['precision']:.4f}")
    print(f"  Recall:    {valid_results[best_nu]['recall']:.4f}")
    print(f"  F1-Score:  {valid_results[best_nu]['f1_score']:.4f}")
    print(f"Anomalies: {true_labels.sum()}")
else:
    # If none 
    best_nu = min(tuning_results, key=lambda x: tuning_results[x]['accuracy'])
    print(f"\nNo model achieved accuracy < 0.90. Using lowest accuracy: {best_nu:.2f}")
    print(f"  Accuracy:  {tuning_results[best_nu]['accuracy']:.4f}")

# -------------------------------
# 4. TRAIN BEST MODEL
# -------------------------------
print(f"\n{'='*50}")
print(f"Training One-Class SVM")
print(f"{'='*50}")

svm_model = OneClassSVM(
    kernel='rbf',
    nu=best_nu,
    gamma='scale',
    tol=1e-3,
    max_iter=-1
)

svm_model.fit(amount_scaled)
predictions_svm = svm_model.predict(amount_scaled)
predictions = np.where(predictions_svm == -1, 1, 0)

# Calculate metrics
accuracy = accuracy_score(true_labels, predictions)
precision = precision_score(true_labels, predictions, zero_division=0)
recall = recall_score(true_labels, predictions, zero_division=0)
f1 = f1_score(true_labels, predictions, zero_division=0)

print(f"Accuracy:  {accuracy:.4f}")
print(f"Precision: {precision:.4f}")
print(f"Recall:    {recall:.4f}")
print(f"F1-Score:  {f1:.4f}")
print(f"Confusion Matrix:")
print(confusion_matrix(true_labels, predictions))

# Get decision function scores
decision_scores = svm_model.decision_function(amount_scaled)

# -------------------------------
# 5. VISUALIZATION
# -------------------------------
fig, axes = plt.subplots(2, 3, figsize=(15, 10))
axes = axes.flatten()

# Plot 1: Original data distribution
ax = axes[0]
ax.hist(amount, bins=50, alpha=0.7, color='blue', edgecolor='black')
ax.axvline(lower_bound, color='red', linestyle='--', label='IQR lower bound')
ax.axvline(upper_bound, color='red', linestyle='--', label='IQR upper bound')
ax.set_title('Original Transaction Amount Distribution')
ax.set_xlabel('Amount')
ax.set_ylabel('Frequency')
ax.legend()

# Plot 2: Data points colored by SVM predictions
ax = axes[1]
colors = ['blue' if p == 0 else 'red' for p in predictions]
ax.scatter(range(len(amount)), amount, c=colors, alpha=0.6, s=20)
ax.set_title(f'SVM Anomaly Detection Results (nu={best_nu:.2f})\nF1={f1:.3f}')
ax.set_xlabel('Transaction Index')
ax.set_ylabel('Amount')

# Plot 3: Performance metrics visualization
ax = axes[2]
metrics = ['Accuracy', 'Precision', 'Recall', 'F1-Score']
values = [accuracy, precision, recall, f1]
bars = ax.bar(metrics, values, color=['#3498db', '#2ecc71', '#e74c3c', '#f39c12'], alpha=0.8)
ax.set_ylim([0, 1.1])
ax.axhline(y=0.90, color='red', linestyle='--', linewidth=2, label='Accuracy < 0.90 Target')
ax.set_ylabel('Score')
ax.set_title('Performance Metrics')
ax.grid(True, alpha=0.3)
ax.legend()
for bar, val in zip(bars, values):
    ax.text(bar.get_x() + bar.get_width()/2, bar.get_height() + 0.02, 
            f'{val:.3f}', ha='center', va='bottom')

# Plot 4: Decision function scores distribution
ax = axes[3]
ax.hist(decision_scores[true_labels==0], bins=30, alpha=0.7, label='Normal', color='blue')
ax.hist(decision_scores[true_labels==1], bins=30, alpha=0.7, label='Anomaly (true)', color='red')
ax.axvline(0, color='green', linestyle='--', linewidth=2, label='Decision Boundary')
ax.set_title('Decision Function Scores Distribution')
ax.set_xlabel('Distance from decision boundary')
ax.set_ylabel('Frequency')
ax.legend()
ax.grid(True, alpha=0.3)

# Plot 5: ROC-style threshold analysis
ax = axes[4]
sorted_scores = np.sort(decision_scores)
cumulative_normal = np.cumsum(true_labels == 0)
cumulative_anomaly = np.cumsum(true_labels == 1)
ax.plot(sorted_scores, cumulative_normal, label='Normal', color='blue', alpha=0.7)
ax.plot(sorted_scores, cumulative_anomaly, label='Anomaly', color='red', alpha=0.7)
ax.axvline(0, color='green', linestyle='--', linewidth=2, label='Decision Boundary')
ax.set_xlabel('Decision Score')
ax.set_ylabel('Cumulative Count')
ax.set_title('Cumulative Distribution by Decision Score')
ax.legend()
ax.grid(True, alpha=0.3)

# Plot 6: Confusion matrix heatmap
ax = axes[5]
cm = confusion_matrix(true_labels, predictions)
im = ax.imshow(cm, cmap='Blues', interpolation='nearest')
ax.set_xticks([0, 1])
ax.set_yticks([0, 1])
ax.set_xticklabels(['Predicted Normal', 'Predicted Anomaly'])
ax.set_yticklabels(['True Normal', 'True Anomaly'])
ax.set_title('Confusion Matrix')
for i in range(2):
    for j in range(2):
        ax.text(j, i, str(cm[i, j]), ha='center', va='center', 
                color='white' if cm[i, j] > cm.max()/2 else 'black')
plt.colorbar(im, ax=ax)

plt.tight_layout()
plt.savefig('svm_anomaly_detection_results.png', dpi=300, bbox_inches='tight')
plt.show()
