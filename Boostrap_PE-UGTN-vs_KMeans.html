class PositionalEncoding(nn.Module):
    def __init__(self, d_model, dropout=0.1, max_len=5000):
        super(PositionalEncoding, self).__init__()
        self.dropout = nn.Dropout(p=dropout)
        pe = torch.zeros(max_len, d_model)
        position = torch.arange(0, max_len, dtype=torch.float).unsqueeze(1)
        div_term = torch.exp(torch.arange(0, d_model, 2).float() * (-np.log(10000.0) / d_model))
        pe[:, 0::2] = torch.sin(position * div_term)
        pe[:, 1::2] = torch.cos(position * div_term)
        pe = pe.unsqueeze(0).transpose(0, 1)
        self.register_buffer('pe', pe)

    def forward(self, x):
        x = x + self.pe[:x.size(0), :]
        return self.dropout(x)

class UnivariateGatedTransformer(nn.Module):
    def __init__(self, input_dim=1, d_model=128, nhead=8, num_layers=3, 
                 dim_feedforward=256, dropout=0.1):
        super(UnivariateGatedTransformer, self).__init__()
        self.input_proj = nn.Linear(input_dim, d_model)
        self.pos_encoder = PositionalEncoding(d_model, dropout)
        self.encoder_layer = nn.TransformerEncoderLayer(
            d_model=d_model, nhead=nhead, dim_feedforward=dim_feedforward, 
            dropout=dropout, batch_first=False
        )
        self.transformer_encoder = nn.TransformerEncoder(
            self.encoder_layer, num_layers=num_layers
        )
        self.fc_forecast = nn.Linear(d_model, input_dim)
        self.fc_reconstruct = nn.Linear(d_model, input_dim)
        self.gate_fc = nn.Linear(d_model, 1)
        self.d_model = d_model
        
    def forward(self, src):
        src = self.input_proj(src)
        src = self.pos_encoder(src)
        latent_seq = self.transformer_encoder(src)
        latent = latent_seq.mean(dim=0)
        gate_weights = torch.sigmoid(self.gate_fc(latent))
        gated_latent = latent * gate_weights
        forecast = self.fc_forecast(gated_latent)
        reconstruction = self.fc_reconstruct(latent_seq)
        return forecast, reconstruction, latent

def compute_reconstruction_errors(model, data_loader, seq_length):
    model.eval()
    recon_errors = []
    with torch.no_grad():
        for i in range(0, len(data_loader), 64):
            batch = data_loader[i:i+64]
            if len(batch) == 0:
                continue
            batch = batch.transpose(0, 1)
            forecast, reconstruction, latent = model(batch)
            recon_loss = ((reconstruction - batch) ** 2).mean(dim=(0, 2))
            recon_errors.extend(recon_loss.cpu().numpy())
    return np.array(recon_errors)

def train_pe_ugtn(model, X_train, y_train, epochs=100, lr=0.001, 
                  recon_weight=1.5, forecast_weight=0.5):
    """
    Optimized training with MAE for reconstruction and MSE for forecasting
    """
    criterion_recon = nn.L1Loss()
    criterion_forecast = nn.MSELoss()
    optimizer = optim.Adam(model.parameters(), lr=lr)
    scheduler = optim.lr_scheduler.ReduceLROnPlateau(optimizer, mode='min', factor=0.5, patience=10)
    
    best_loss = float('inf')
    
    for epoch in range(epochs):
        model.train()
        optimizer.zero_grad()
        forecast, reconstruction, latent = model(X_train.transpose(0, 1))
        recon_loss = criterion_recon(reconstruction, X_train.transpose(0, 1))
        forecast_loss = criterion_forecast(forecast.squeeze(), y_train.squeeze())
        
        loss = recon_weight * recon_loss + forecast_weight * forecast_loss
        
        loss.backward()
        torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
        optimizer.step()
        
        # Validation
        model.eval()
        with torch.no_grad():
            val_forecast, val_reconstruction, _ = model(X_val_t.transpose(0, 1))
            val_recon_loss = criterion_recon(val_reconstruction, X_val_t.transpose(0, 1))
            val_forecast_loss = criterion_forecast(val_forecast.squeeze(), y_val_t.squeeze())
            val_loss = recon_weight * val_recon_loss + forecast_weight * val_forecast_loss
        
        scheduler.step(val_loss)
        
        if val_loss < best_loss:
            best_loss = val_loss
            best_model_state = model.state_dict()
        
        if (epoch+1) % 20 == 0:
            print(f"  Epoch {epoch+1}: Recon Loss = {recon_loss.item():.4f}, "
                  f"Forecast Loss = {forecast_loss.item():.4f}, "
                  f"Val Loss = {val_loss.item():.4f}")
    
    model.load_state_dict(best_model_state)
    return model

class TrimmedKMeans:
    """
    Trimmed K-Means clustering algorithm that trims a percentage of points
    considered as outliers/anomalies.
    """
    def __init__(self, n_clusters=3, trim_percentage=0.1, max_iter=100, random_state=42):
        self.n_clusters = n_clusters
        self.trim_percentage = trim_percentage
        self.max_iter = max_iter
        self.random_state = random_state
        self.centroids = None
        self.labels_ = None
        self.trimmed_indices = None
        
    def fit(self, X):
        np.random.seed(self.random_state)
        n_samples = X.shape[0]
        n_trim = int(n_samples * self.trim_percentage)
        
        # Initialize centroids randomly
        indices = np.random.choice(n_samples, self.n_clusters, replace=False)
        self.centroids = X[indices].copy()
        
        # Keep track of which points are kept (not trimmed)
        kept_indices = np.ones(n_samples, dtype=bool)
        
        for iteration in range(self.max_iter):
            # Calculate distances to centroids for all points
            distances = np.zeros((n_samples, self.n_clusters))
            for k in range(self.n_clusters):
                distances[:, k] = np.linalg.norm(X - self.centroids[k], axis=1)
            
            # Assign each point to nearest centroid
            assignments = np.argmin(distances, axis=1)
            min_distances = np.min(distances, axis=1)
            
            # Trim points with largest distances to their assigned centroid
            trim_threshold = np.sort(min_distances)[-n_trim] if n_trim > 0 else float('inf')
            trim_mask = min_distances > trim_threshold
            
            # If using fixed percentage, ensure exactly n_trim points are trimmed
            if n_trim > 0:
                trim_indices = np.argsort(min_distances)[-n_trim:]
                trim_mask = np.zeros(n_samples, dtype=bool)
                trim_mask[trim_indices] = True
            
            self.trimmed_indices = np.where(trim_mask)[0]
            kept_indices = ~trim_mask
            
            # Update centroids using only kept points
            new_centroids = np.zeros_like(self.centroids)
            for k in range(self.n_clusters):
                cluster_points = X[kept_indices & (assignments == k)]
                if len(cluster_points) > 0:
                    new_centroids[k] = cluster_points.mean(axis=0)
                else:
                    # If cluster is empty, reinitialize
                    new_centroids[k] = X[np.random.choice(n_samples)]
            
            # Check for convergence
            if np.allclose(self.centroids, new_centroids, atol=1e-4):
                break
                
            self.centroids = new_centroids
        
        # Final assignment and trimming
        distances = np.zeros((n_samples, self.n_clusters))
        for k in range(self.n_clusters):
            distances[:, k] = np.linalg.norm(X - self.centroids[k], axis=1)
        
        self.labels_ = np.argmin(distances, axis=1)
        min_distances = np.min(distances, axis=1)
        
        if n_trim > 0:
            trim_indices = np.argsort(min_distances)[-n_trim:]
            self.trimmed_indices = trim_indices
        
        return self
    
    def predict(self, X):
        """Predict cluster assignments for new data."""
        distances = np.zeros((X.shape[0], self.n_clusters))
        for k in range(self.n_clusters):
            distances[:, k] = np.linalg.norm(X - self.centroids[k], axis=1)
        return np.argmin(distances, axis=1)
    
    def get_anomaly_scores(self, X):
        """Get anomaly scores based on distance to nearest centroid."""
        distances = np.zeros((X.shape[0], self.n_clusters))
        for k in range(self.n_clusters):
            distances[:, k] = np.linalg.norm(X - self.centroids[k], axis=1)
        return np.min(distances, axis=1)

def bootstrap_stability_comparison(pe_scores, kmeans_scores, k=100, n_iterations=200, sample_size=0.7):
    """
    Compare PE-UGTN and Trimmed K-Means using bootstrap stability analysis.
    """
    
    n_total = len(pe_scores)
    n_sample = int(n_total * sample_size)
    
    # Storage for top-k indices
    pe_top_k_lists = []
    kmeans_top_k_lists = []
    cross_agreement = []
    spearman_correlations = []
    
    print(f"\nRunning {n_iterations} bootstrap iterations...")
    
    for i in tqdm(range(n_iterations)):
        # Sample indices with replacement
        sampled_indices = np.random.choice(n_total, size=n_sample, replace=True)
        
        # Get scores for sampled data
        pe_sample = pe_scores[sampled_indices]
        kmeans_sample = kmeans_scores[sampled_indices]
        
        # Get top-k indices (highest scores = most anomalous)
        pe_top_k = np.argsort(pe_sample)[-k:]
        kmeans_top_k = np.argsort(kmeans_sample)[-k:]
        
        # Store original indices
        pe_top_k_lists.append(set(sampled_indices[pe_top_k]))
        kmeans_top_k_lists.append(set(sampled_indices[kmeans_top_k]))
        
        # Cross-model agreement (Jaccard on this iteration)
        intersection = len(pe_top_k_lists[-1] & kmeans_top_k_lists[-1])
        union = len(pe_top_k_lists[-1] | kmeans_top_k_lists[-1])
        cross_agreement.append(intersection / union if union > 0 else 0)
        
        # Spearman rank correlation on this iteration
        from scipy.stats import spearmanr
        corr, _ = spearmanr(pe_sample, kmeans_sample)
        spearman_correlations.append(corr if not np.isnan(corr) else 0)
    
    # Calculate stability for each model
    def calculate_stability(top_k_lists):
        stabilities = []
        n_iter = len(top_k_lists)
        for i in range(n_iter):
            for j in range(i+1, n_iter):
                intersection = len(top_k_lists[i] & top_k_lists[j])
                union = len(top_k_lists[i] | top_k_lists[j])
                if union > 0:
                    stabilities.append(intersection / union)
        return np.mean(stabilities) if stabilities else 0, np.std(stabilities) if stabilities else 0
    
    pe_stability_mean, pe_stability_std = calculate_stability(pe_top_k_lists)
    kmeans_stability_mean, kmeans_stability_std = calculate_stability(kmeans_top_k_lists)
    
    # Results
    results = {
        'pe_ugtn_stability_mean': pe_stability_mean,
        'pe_ugtn_stability_std': pe_stability_std,
        'kmeans_stability_mean': kmeans_stability_mean,
        'kmeans_stability_std': kmeans_stability_std,
        'cross_model_agreement_mean': np.mean(cross_agreement),
        'cross_model_agreement_std': np.std(cross_agreement),
        'spearman_correlation_mean': np.mean(spearman_correlations),
        'spearman_correlation_std': np.std(spearman_correlations),
        'pe_top_k_lists': pe_top_k_lists,
        'kmeans_top_k_lists': kmeans_top_k_lists,
        'cross_agreement': cross_agreement,
        'spearman_correlations': spearman_correlations
    }
    
    return results

class PositionalEncoding(nn.Module):
    def __init__(self, d_model, dropout=0.1, max_len=5000):
        super(PositionalEncoding, self).__init__()
        self.dropout = nn.Dropout(p=dropout)
        pe = torch.zeros(max_len, d_model)
        position = torch.arange(0, max_len, dtype=torch.float).unsqueeze(1)
        div_term = torch.exp(torch.arange(0, d_model, 2).float() * (-np.log(10000.0) / d_model))
        pe[:, 0::2] = torch.sin(position * div_term)
        pe[:, 1::2] = torch.cos(position * div_term)
        pe = pe.unsqueeze(0).transpose(0, 1)
        self.register_buffer('pe', pe)

    def forward(self, x):
        x = x + self.pe[:x.size(0), :]
        return self.dropout(x)

class UnivariateGatedTransformer(nn.Module):
    def __init__(self, input_dim=1, d_model=128, nhead=8, num_layers=3, 
                 dim_feedforward=256, dropout=0.1):
        super(UnivariateGatedTransformer, self).__init__()
        self.input_proj = nn.Linear(input_dim, d_model)
        self.pos_encoder = PositionalEncoding(d_model, dropout)
        self.encoder_layer = nn.TransformerEncoderLayer(
            d_model=d_model, nhead=nhead, dim_feedforward=dim_feedforward, 
            dropout=dropout, batch_first=False
        )
        self.transformer_encoder = nn.TransformerEncoder(
            self.encoder_layer, num_layers=num_layers
        )
        self.fc_forecast = nn.Linear(d_model, input_dim)
        self.fc_reconstruct = nn.Linear(d_model, input_dim)
        self.gate_fc = nn.Linear(d_model, 1)
        self.d_model = d_model
        
    def forward(self, src):
        src = self.input_proj(src)
        src = self.pos_encoder(src)
        latent_seq = self.transformer_encoder(src)
        latent = latent_seq.mean(dim=0)
        gate_weights = torch.sigmoid(self.gate_fc(latent))
        gated_latent = latent * gate_weights
        forecast = self.fc_forecast(gated_latent)
        reconstruction = self.fc_reconstruct(latent_seq)
        return forecast, reconstruction, latent

class TrimmedKMeans:
    """
    Trimmed K-Means clustering algorithm that trims a percentage of points
    considered as outliers/anomalies.
    """
    def __init__(self, n_clusters=3, trim_percentage=0.1, max_iter=100, random_state=42):
        self.n_clusters = n_clusters
        self.trim_percentage = trim_percentage
        self.max_iter = max_iter
        self.random_state = random_state
        self.centroids = None
        self.labels_ = None
        self.trimmed_indices = None
        
    def fit(self, X):
        np.random.seed(self.random_state)
        n_samples = X.shape[0]
        n_trim = int(n_samples * self.trim_percentage)
        
        # Initialize centroids randomly
        indices = np.random.choice(n_samples, self.n_clusters, replace=False)
        self.centroids = X[indices].copy()
        
        # Keep track of which points are kept (not trimmed)
        kept_indices = np.ones(n_samples, dtype=bool)
        
        for iteration in range(self.max_iter):
            # Calculate distances to centroids for all points
            distances = np.zeros((n_samples, self.n_clusters))
            for k in range(self.n_clusters):
                distances[:, k] = np.linalg.norm(X - self.centroids[k], axis=1)
            
            # Assign each point to nearest centroid
            assignments = np.argmin(distances, axis=1)
            min_distances = np.min(distances, axis=1)
            
            # Trim points with largest distances to their assigned centroid
            trim_threshold = np.sort(min_distances)[-n_trim] if n_trim > 0 else float('inf')
            trim_mask = min_distances > trim_threshold
            
            # If using fixed percentage, ensure exactly n_trim points are trimmed
            if n_trim > 0:
                trim_indices = np.argsort(min_distances)[-n_trim:]
                trim_mask = np.zeros(n_samples, dtype=bool)
                trim_mask[trim_indices] = True
            
            self.trimmed_indices = np.where(trim_mask)[0]
            kept_indices = ~trim_mask
            
            # Update centroids using only kept points
            new_centroids = np.zeros_like(self.centroids)
            for k in range(self.n_clusters):
                cluster_points = X[kept_indices & (assignments == k)]
                if len(cluster_points) > 0:
                    new_centroids[k] = cluster_points.mean(axis=0)
                else:
                    # If cluster is empty, reinitialize
                    new_centroids[k] = X[np.random.choice(n_samples)]
            
            # Check for convergence
            if np.allclose(self.centroids, new_centroids, atol=1e-4):
                break
                
            self.centroids = new_centroids
        
        # Final assignment and trimming
        distances = np.zeros((n_samples, self.n_clusters))
        for k in range(self.n_clusters):
            distances[:, k] = np.linalg.norm(X - self.centroids[k], axis=1)
        
        self.labels_ = np.argmin(distances, axis=1)
        min_distances = np.min(distances, axis=1)
        
        if n_trim > 0:
            trim_indices = np.argsort(min_distances)[-n_trim:]
            self.trimmed_indices = trim_indices
        
        return self
    
    def predict(self, X):
        """Predict cluster assignments for new data."""
        distances = np.zeros((X.shape[0], self.n_clusters))
        for k in range(self.n_clusters):
            distances[:, k] = np.linalg.norm(X - self.centroids[k], axis=1)
        return np.argmin(distances, axis=1)
    
    def get_anomaly_scores(self, X):
        """Get anomaly scores based on distance to nearest centroid."""
        distances = np.zeros((X.shape[0], self.n_clusters))
        for k in range(self.n_clusters):
            distances[:, k] = np.linalg.norm(X - self.centroids[k], axis=1)
        return np.min(distances, axis=1)

def bootstrap_stability_comparison(pe_scores, kmeans_scores, k=100, n_iterations=200, sample_size=0.7):

        sampled_indices = np.random.choice(n_total, size=n_sample, replace=True)

        from scipy.stats import spearmanr
        corr, _ = spearmanr(pe_sample, kmeans_sample)
        spearman_correlations.append(corr if not np.isnan(corr) else 0)
    
    # Calculate stability for each model
    def calculate_stability(top_k_lists):
        stabilities = []
        n_iter = len(top_k_lists)
        for i in range(n_iter):
            for j in range(i+1, n_iter):
                intersection = len(top_k_lists[i] & top_k_lists[j])
                union = len(top_k_lists[i] | top_k_lists[j])
                if union > 0:
                    stabilities.append(intersection / union)
        return np.mean(stabilities) if stabilities else 0, np.std(stabilities) if stabilities else 0
    
    pe_stability_mean, pe_stability_std = calculate_stability(pe_top_k_lists)
    kmeans_stability_mean, kmeans_stability_std = calculate_stability(kmeans_top_k_lists)
    
    # Results
    results = {
        'pe_ugtn_stability_mean': pe_stability_mean,
        'pe_ugtn_stability_std': pe_stability_std,
        'kmeans_stability_mean': kmeans_stability_mean,
        'kmeans_stability_std': kmeans_stability_std,
        'cross_model_agreement_mean': np.mean(cross_agreement),
        'cross_model_agreement_std': np.std(cross_agreement),
        'spearman_correlation_mean': np.mean(spearman_correlations),
        'spearman_correlation_std': np.std(spearman_correlations),
        'pe_top_k_lists': pe_top_k_lists,
        'kmeans_top_k_lists': kmeans_top_k_lists,
        'cross_agreement': cross_agreement,
        'spearman_correlations': spearman_correlations
    }
    
    return results
